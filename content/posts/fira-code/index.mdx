pyftsubset FiraCode-VF.woff2 \
  --unicodes="U+20-7E" \
  --layout-features="salt,calt,case,cpsp,ss01,ss02,ss03,ss04,ss05,ss06" \
  --flavor="woff2" \
  --output-file="FiraCode-VF-subset.woff2"
